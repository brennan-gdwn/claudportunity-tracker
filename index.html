Browser Extension Setup

Follow these steps to set up the browser extension that will help you capture opportunities from any website:

1 Install a browser extension that allows for custom scripts, like Tampermonkey (Chrome) or Greasemonkey (Firefox).
2 Create a new script and paste the following code:
// ==UserScript== // @name Opportunity Tracker Extension // @namespace http://tampermonkey.net/ // @version 0.1 // @description Capture professional opportunities from any website // @author You // @match *://*/* // @grant GM_registerMenuCommand // @grant GM_setValue // @grant GM_getValue // ==/UserScript== (function() { 'use strict'; // Register a command in the Tampermonkey menu GM_registerMenuCommand("Capture Opportunity", captureOpportunity); GM_registerMenuCommand("Capture Contact", captureContact); GM_registerMenuCommand("Capture Note", captureNote); function captureOpportunity() { const selection = window.getSelection().toString().trim(); const url = window.location.href; const companyName = prompt("Company Name:", getCompanyFromPage()); if (!companyName) return; const role = prompt("Role/Position:", getSelectionOrDefault(selection, "")); const type = prompt("Type (vc, startup, corporate, nonprofit):", ""); const region = prompt("Region (latam, africa, india, europe, namerica, asia):", ""); const notes = prompt("Notes:", selection); const opportunity = { id: Date.now().toString(), company: companyName, role: role, type: type, region: region, status: "new", notes: notes, url: url, dateAdded: new Date().toISOString(), lastUpdated: new Date().toISOString() }; // Get existing opportunities or initialize empty array const opportunities = JSON.parse(localStorage.getItem('opportunities') || '[]'); opportunities.push(opportunity); localStorage.setItem('opportunities', JSON.stringify(opportunities)); alert("Opportunity captured successfully!"); } function captureContact() { const selection = window.getSelection().toString().trim(); const url = window.location.href; const name = prompt("Contact Name:", getSelectionOrDefault(selection, "")); if (!name) return; const company = prompt("Company:", getCompanyFromPage()); const role = prompt("Role/Position:", ""); const email = prompt("Email:", ""); const phone = prompt("Phone:", ""); const notes = prompt("Notes:", ""); const contact = { id: Date.now().toString(), name: name, company: company, role: role, email: email, phone: phone, notes: notes, url: url, dateAdded: new Date().toISOString(), lastUpdated: new Date().toISOString() }; // Get existing contacts or initialize empty array const contacts = JSON.parse(localStorage.getItem('contacts') || '[]'); contacts.push(contact); localStorage.setItem('contacts', JSON.stringify(contacts)); alert("Contact captured successfully!"); } function captureNote() { const selection = window.getSelection().toString().trim(); const url = window.location.href; const title = prompt("Note Title:", getPageTitle()); if (!title) return; const content = prompt("Note Content:", selection); const tags = prompt("Tags (comma separated):", ""); const note = { id: Date.now().toString(), title: title, content: content, tags: tags.split(',').map(tag => tag.trim()).filter(tag => tag), url: url, dateAdded: new Date().toISOString(), lastUpdated: new Date().toISOString() }; // Get existing notes or initialize empty array const notes = JSON.parse(localStorage.getItem('notes') || '[]'); notes.push(note); localStorage.setItem('notes', JSON.stringify(notes)); alert("Note captured successfully!"); } // Helper functions function getCompanyFromPage() { // Try to extract company name from URL or page content // This is a simple implementation and might need customization const hostname = window.location.hostname; const domainParts = hostname.split('.'); if (hostname.includes('linkedin.com')) { // Try to get company name from LinkedIn page const companyElement = document.querySelector('.org-top-card-summary__title'); if (companyElement) return companyElement.textContent.trim(); } if (domainParts.length > 1) { return domainParts[domainParts.length - 2].charAt(0).toUpperCase() + domainParts[domainParts.length - 2].slice(1); } return ""; } function getPageTitle() { return document.title; } function getSelectionOrDefault(selection, defaultValue) { return selection || defaultValue; } })();
3 Save the script and enable it in your extension.
4 Now you can capture opportunities from any webpage:
  • Right-click on the webpage and go to the Tampermonkey/Greasemonkey menu
  • Select "Capture Opportunity", "Capture Contact", or "Capture Note"
  • Fill in the prompted information
  • The data will be stored locally in your browser
5 To transfer data to this app, use the Export function in the extension and Import function here.
